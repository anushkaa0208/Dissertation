##########
#Point Estimation

set.seed(12345)
n=1000
x=seq(1,n,1)
p=array(0)

for(k in 1:1000)
{
  m=0
  for(i in 1:2000)
  {
    y=sample(x,n,replace=FALSE) #sampling without replacement
    a=0
    for(j in 1:n)
       {
         if(x[j]==y[j])
         {
           a=a+1;
         }
      }
      if(a==0)
      {
         m=m+1
      }
   }
   p[k]=m/2000
}
p #Array of 1000 estimates of e^{-1}


#Bias
bias_e=mean(p-0.367879)

#Mean
mean_e=mean(p)

#MSE
MSE_e=mean((p-0.367879)^2)

#Variance
Var_e=mean((p-mean(p))^2)

#SD
SD_e=sqrt(mean((p-mean(p))^2))

#CV
CV_e=mean_e/SD_e
